129 

130 # Clean up images input 

131 if isinstance(images, (Path, str)):

136 # Sheet-Components predictions 

137 component_files = self.sheet_component_detect(images, output_dir=output_dir)

139 ocr_data = {} 

140 

141 # Institutional Label Field Detection Model Predictions 

142 for stub, components in component_files.items(): 

143 for component in components: 

144 if component.name.endswith("institutional_label.jpg"): 

145 ocr_data[str(component)] = self.institutional_label_detect( 

146 component, 

147 stub=stub, 

148 output_dir=output_dir / stub, 

149 ) 

150 

151 df = ocr_data_df(ocr_data, output_path=output_dir/"ocr_results.csv") 

152 

153 # Write report 

154 if report: 

155 if len(images) == 1: 

156 # If report is for a single image, include in report name 

157 report_path = output_dir/f"hespi-report-{images[0].name}.html" 

158 else: 

159 # If report is for multiple images, make sure it is unique 

160 report_path = output_dir/"hespi-report.html" 

161 index = 1 

162 while report_path.exists(): 

163 index += 1 

164 report_path = output_dir/f"hespi-report-{index}.html" 

165 

166 write_report(report_path, component_files, df) 

167

225 def read_field_file( 

226 self, 

227 field_file:Path, 

228 classification:str = None, 

229 ) -> Dict: 

230 """Reads the text of an image of a field from an institutional label. 

231 

232 Args: 

233 field_file (Path): The path to an image file of a text field.  

234 The type of field needs to be in the second last component of the filename when split by periods. 

235 e.g. XXXXX.FIELD_NAME.jpg 

236 ^^^^^^^^^^ 

237 classification (str, optional): The classification of the institutional label from the institutional_label_classifier model. 

238 If it is 'printed' or 'typewriter' then the Tesseract OCR result will be favoured. 

239 Otherwise the TrOCR model result will be favoured. 

240 

241 Returns: 

242 Dict: A dictionary with the results from TrOCR (if the `htr` flag is on), Tesseract and an adjusted form of the text, 

243 which changes the case depending on the field type and fuzzy matched with the reference database if `fuzzy` is requested. 

244 """ 

245 field_file = Path(field_file) 

246 console.print("field_file:", field_file) 

247 field_file_components = field_file.name.split(".") 

248 assert len(field_file_components) >= 2 

249 field = field_file_components[-2] 

250 classification = classification or "" 

251 

252 detection_results = {} 

253 

254 

255 detection_results[f"{field}_image"] = field_file 

256 

257 # HTR 

258 recognised_text = "" 

259 if self.htr: 

260 htr_text = self.trocr.get_text(field_file) 

261 if htr_text: 

262 print(f"HTR: {htr_text}") 

263 console.print( 

264 f"Handwritten Text Recognition (TrOCR): [red]'{htr_text}'[/red]" 

265 ) 

266 

267 detection_results[f"{field}_TrOCR"] = htr_text 

268 recognised_text = htr_text 

269 

270 # OCR 

271 tesseract_text = self.tesseract.get_text(field_file) 

272 if tesseract_text: 

273 detection_results[f"{field}_Tesseract"] = tesseract_text 

274 console.print( 

275 f"Optical Character Recognition (Tesseract): [red]'{tesseract_text}'[/red]" 

276 ) 

277 

278 if classification in ["printed", "typewriter"] or not recognised_text: 

279 recognised_text = tesseract_text 

280 

281 # Adjust text if necessary 

282 if recognised_text: 

283 detection_results[field] = adjust_text( 

284 field, 

285 recognised_text, 

286 fuzzy=self.fuzzy, 

287 fuzzy_cutoff=self.fuzzy_cutoff, 

288 reference=self.reference, 

289 ) 

290 

291 return detection_results
