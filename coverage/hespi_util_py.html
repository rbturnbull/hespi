95def ocr_data_df(data: dict, output_path: Path=None) -> pd.DataFrame: 

96 """  

97 Creates a DataFrame of data, sorts columns and outputs a CSV with OCR values. 

98 

99 Args: 

100 data (dict): The data coming from the text recognition models.  

101 The keys are the institutional labels. 

102 The values are dictionaries with the fields as keys and the values as the recognised text. 

103 output_path (Path, optional): Where to save a CSV file if given. Defaults to None. 

104 

105 Returns: 

106 pd.DataFrame: The text recognition data as a Pandas dataframe 

107 """ 

108 df = pd.DataFrame.from_dict(data, orient="index") 

109 df = df.fillna(value="") 

110 df = df.reset_index().rename(columns={"index": "institutional label"}) 

111 

112 # Splitting the ocr_results columns into seperate original text, adjusted, and score 

113 # Enables the html report to pull the data  

114 for col in df.columns: 

115 if 'ocr_results' in col: 

116 field_name = col.replace('_ocr_results', '') 

117 new_columns = df[f"{field_name}_ocr_results"].apply(process_row_ocr_results, field_name=field_name).apply(pd.Series) 

118 

119 df = pd.concat([df, new_columns], axis=1) 

120 

121 # insert columns not included in dataframe, and re-order 

122 # including any columns not included in col_options to account for any updates 

123 col_options = [ "institutional label", "id" ] + label_fields 

124 

125 missing_cols = [col for col in col_options if col not in df.columns] 

126 df[missing_cols] = "" 

127 

128 # creating break columns 

129 df['<--results|ocr_details-->'] = ' | ' 

130 df['image_links-->'] = ' | ' 

131 df['ocr_results_split-->'] = ' | ' 

132 

133 # grouping other columns  

134 score_cols = sorted([col for col in df.columns if '_match_score' in col and 'Tesseract' not in col and 'TrOCR' not in col], key=label_sort_key) 

135 ocr_cols = ['<--results|ocr_details-->'] + sorted([col for col in df.columns if '_ocr_results' in col], key=label_sort_key) 

136 image_files_cols = ['image_links-->'] + sorted([col for col in df.columns if '_image' in col or 'predictions' in col], key=label_sort_key) 

137 result_cols = ['ocr_results_split-->'] + sorted([col for col in df.columns if 'Tesseract' in col or 'TrOCR' in col], key=label_sort_key) 

138 

139 cols = col_options + score_cols + ['label_classification'] + ocr_cols + image_files_cols + result_cols 

140 

141 extra_cols = [col for col in df.columns if col not in cols] 

142 

143 cols = cols + extra_cols 

144 

145 # Filter for only the columns that are in the dataframe 

146 cols = [col for col in cols if col in df.columns] 

147 

148 df = df[cols] 

149 df = df.fillna('') 

150 

151 # flattening all the lists so that if only one item, the list is removed, if no items, list is replaced with an empty string 

152 for col in df.columns: 

153 df[col] = df[col].apply(flatten_single_item_lists) 

154 

155 # CSV output 

156 if output_path: 

157 output_path = Path(output_path) 

158 output_path.parent.mkdir(exist_ok=True, parents=True) 

159 

160 # If the file already exists, then concatenate it 

161 if output_path.exists(): 

162 old_df = pd.read_csv(output_path) 

163 write_df = pd.concat([old_df, df]) 

164 console.print(f"Appending Hespi text results to: '{output_path}'") 

165 else: 

166 console.print(f"Writing Hespi text results to: '{output_path}'") 

167 write_df = df 

168 

169 write_df.to_csv(output_path, index=False) 

170 

171 return df

174def ocr_result_str(row, field_name:str, ocr_type:str) -> str: 

175 def get_list(row, key): 

176 result = row.get(key, "") 

177 if not isinstance(result, list): 

178 result = [result] 

179 

180 return result 

181 

182 original_list = get_list(row, f"{field_name}_{ocr_type}_original") 

183 adjusted_list = get_list(row, f"{field_name}_{ocr_type}_adjusted") 

184 match_score_list = get_list(row, f"{field_name}_{ocr_type}_match_score") 

185 

186 assert len(original_list) == len(adjusted_list) == len(match_score_list) 

187 

188 texts = [] 

189 for original, adjusted, match_score in zip(original_list, adjusted_list, match_score_list): 

190 text = original 

191 if adjusted and adjusted != original: 

192 text += f" → {adjusted}" 

193 

194 if match_score: 

195 text += f" ({match_score})" 

196 

197 texts.append(text) 

198 

199 return " | ".join(texts)

225def adjust_text(field:str, recognised_text:str, fuzzy:bool, fuzzy_cutoff:float, reference:Dict): 

226 text_adjusted = adjust_case(field, recognised_text) 

227 match_score = "" 

228 

229 # Match with database 

230 if fuzzy and field in reference: 

231 close_matches = get_close_matches( 

232 text_adjusted, 

233 reference[field], 

234 cutoff=fuzzy_cutoff, 

235 n=1, 

236 ) 

237 if close_matches: 

238 match_score = round(SequenceMatcher(None, text_adjusted, close_matches[0]).ratio(), 3) 

239 text_adjusted = close_matches[0] 

240 else: 

241 match_score = 0 

242 

243 return (text_adjusted, match_score)
