149def csv_creation(data:dict, output_dir:Path): 

150 """ 

151 Creates a DataFrame of data, checks if OCR output is a known value, and outputs a CSV with OCR values 

152 """ 

153 

154 df = pd.DataFrame.from_dict(data, orient='index') 

155 df = df.reset_index().rename(columns={"index": "institutional label"}) 

156 

157 # insert columns not included in dataframe, and re-order 

158 # including any columns not included in col_options to account for any updates 

159 col_options = ['institutional label', "id", 'family', 'genus', 'species', 'infrasp taxon', 

160 'authority', 'collector_number', 'collector', 

161 'locality', 'geolocation', 'year', 'month', 'day'] 

162 

163 missing_cols = [col for col in col_options if col not in df.columns] 

164 df[missing_cols] = '' 

165 extra_cols = [col for col in df.columns if col not in col_options] 

166 cols = col_options + extra_cols 

167 df = df[cols] 

168 

169 # CSV output 

170 df.to_csv(str(output_dir)+'/ocr_results.csv', index=False) 

171 return df
